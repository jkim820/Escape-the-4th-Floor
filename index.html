* This function numberOfPrimes(n), takes in a number, `n` and returns
the number of prime numbers less than or equal to `n`.
* The answer to this challenge will be used to solve the final question.

function isPrime?(num) {
    if ( num < 2 ) return false;
    for ( let i = 2; i < num; i++ ) {
       if ( num % i === 0 ) return false;
    }
    return true;
}

function numberOfPrimes(n) {
    let numPrimes = 0;
    for ( let i = 2; i <= n; i++ ) {
       if ( isPrime?(i) ) numPrimes += 1;
    }
    return numPrimes;
}

def first_letters(string)
    new_string = [ ]
    words = string.split(" ")
    words.each do |word|
        new_string << word[0]
    end

    return new_string.join("")
end
